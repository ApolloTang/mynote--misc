
=============================
  Loop over same indexes but
  order order is different
=============================

  let C = 3;
  let i = 0;
  for (i; i<C; i++) {
    // 0, 1, 2
  }

  let C = 3;
  let i = C;
  while (i--) {
    // 2, 1, 0
  }

  More While loop

  let i = 0;
  while( i++ < 3) {
    // i >>> 1, 2, 3
  };

  let j = 0;
  while( ++j < 3) {
    // j >>> 1, 2
  };
